{
    int i,n1,n2,f=0,count=0;

while(n1<=n2)
{
    for(i=1;i<=n1;i++)
    {
        if(n1%i==0)
        f++;
    }
    if(f==2)

        printf("%d is a prime number\n",n1);

    }

f=0;
